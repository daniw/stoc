  \item Modell: \\
        \[ X \sim \text{Bin}(n=10,\pi) \]
  \item Nullhypothese: 
        \[ \begin{array}{ll}
          H_0 &: \pi_0 = 0.3 \\
          H_A &: \pi > \pi_0
        \end{array} \]
  \item Teststatistik: 
        \[ X: P(X=x | H_0) 
        = \left( \begin{array}{l}10\\x\end{array} \right) 
        \cdot 0.3^x \cdot 0.7^{10 - x} \]
  \item $\alpha = 0.05$
  \item $P(X=x)$ in Abhängigkeit von x:
